Crypto Price Tracker

document.getElementById('crypto-form').addEventListener('submit', async function (e) { e.preventDefault(); const cryptoName = document.getElementById('crypto-name').value; const priceDisplay = document.getElementById('price-display'); const alertSound = document.getElementById('alert-sound'); priceDisplay.innerHTML = `

Loading...

`; alertSound.pause(); alertSound.currentTime = 0; try { const response = await fetch(`https://api.bybit.com/v2/public/tickers?symbol=${cryptoName}`); const data = await response.json(); if (data.result && data.result.length > 0) { const price = data.result[0].last_price; priceDisplay.innerHTML = `

${cryptoName} Price: $${price}

`; alertSound.play(); } else { priceDisplay.innerHTML = `

Invalid cryptocurrency name or not available on Bybit.

`; } } catch (error) { priceDisplay.innerHTML = `

Error fetching price.

`; } });
